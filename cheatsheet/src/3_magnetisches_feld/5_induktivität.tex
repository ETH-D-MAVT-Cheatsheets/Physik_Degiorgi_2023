\subsection{Induktivität L}
    \begin{minipage}{0.49\linewidth}
        \begin{empheq}[box = \fbox]{align*}
            L_i &= \frac{n \Phi}{I}\\
            &= \mu n^2 \frac{A}{l} \text{dünne Spule}
        \end{empheq}
        Serie: $L_{tot} = \sum\limits_i L_i$\\
        Parallel: $\frac{1}{L_{tot}} = \sum\limits_i \frac{1}{L_i}$
    \end{minipage}
    \begin{minipage}{0.49\linewidth}
        \begin{scriptsize}
            \begin{empheq}{align*}
                \mu &= \mu_0 \cdot \mu_r = \text{Mag. Permeabilität}\\
                \Phi &= \text{Magnetischer Fluss durch die Spule}\\
                n &= \text{Windungen der Spule}\\
                l &= \text{Länge der Spule}\\
                A_i &= \text{Querschnittsfläche der Spule}\\
            \end{empheq}
        \end{scriptsize}
    \end{minipage}
