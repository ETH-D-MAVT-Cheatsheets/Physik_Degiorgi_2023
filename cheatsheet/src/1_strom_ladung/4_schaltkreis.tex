                \begin{align*}
                    \frac{1}{R_{res}} &= \sum \frac{1}{R_i}
                    \\\frac{1}{R_{res}} &= \frac{1}{R_1} +\frac{1}{R_2}
                    \\\text{Nur für zwei Widerstände:}
                    \\R_{res} &= \frac{R_1 \cdot R_2}{R_1 + R_2}
                \end{align*}
            \end{center}
        \end{scriptsize}
